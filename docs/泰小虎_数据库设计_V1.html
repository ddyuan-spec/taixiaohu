泰小虎 - 数据库设计文档 V1.0

文档说明：本文档定义泰小虎智能健康导购助手的核心数据库表结构，支持用户画像、购买记录、关联人管理、客服配置等业务场景。
> 版本：V1.0
创建日期：2026-05-25
所属项目：泰小虎 - 智能健康导购助手

一、数据库概述

1.1 数据库选型

类型数据库用途
关系型数据库MySQL 8.0 / PostgreSQL用户数据、订单数据、配置数据
缓存数据库Redis会话缓存、热点数据
向量数据库Volcano Engine RAG知识库检索（Coze内置）

1.2 表结构总览

序号表名中文名说明
1t_user用户表存储用户基础信息
2t_user_profile用户画像表存储用户健康画像
3t_user_relation关联人表存储用户关联的家人信息
4t_relation_profile关联人画像表存储关联人健康画像
5t_purchase_record购买记录表存储用户购买历史
6t_profile_tag画像标签表存储可维护的画像标签定义
7t_profile_change_log画像变更记录表记录画像重大变更
8t_customer_service客服配置表存储客服联系方式配置
9t_conversation_history对话历史表存储用户对话记录

二、核心表结构设计

2.1 用户表 (t_user)

CREATE TABLE t_user (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '用户ID',
    phone VARCHAR(20) NOT NULL UNIQUE COMMENT '手机号',
    nickname VARCHAR(50) COMMENT '昵称',
    avatar VARCHAR(255) COMMENT '头像URL',
    gender TINYINT COMMENT '性别：0-未知，1-男，2-女',
    birth_date DATE COMMENT '出生日期',
    status TINYINT DEFAULT 1 COMMENT '状态：0-禁用，1-正常',
    last_login_time DATETIME COMMENT '最后登录时间',
    last_login_device VARCHAR(100) COMMENT '最后登录设备',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    deleted_at DATETIME COMMENT '删除时间（软删除）',
    INDEX idx_phone (phone),
    INDEX idx_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

2.2 用户画像表 (t_user_profile)

CREATE TABLE t_user_profile (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '画像ID',
    user_id BIGINT NOT NULL UNIQUE COMMENT '用户ID',
    
    -- 基础健康信息
    height DECIMAL(5,2) COMMENT '身高(cm)',
    weight DECIMAL(5,2) COMMENT '体重(kg)',
    blood_type VARCHAR(5) COMMENT '血型：A/B/AB/O',
    
    -- 健康状况
    chronic_diseases JSON COMMENT '慢性病史，如["高血压","糖尿病"]',
    allergies JSON COMMENT '过敏史，如["青霉素","海鲜"]',
    current_medications JSON COMMENT '当前用药情况',
    health_concerns JSON COMMENT '健康关注点，如["骨骼","睡眠","心脑血管"]',
    
    -- 生活习惯
    exercise_frequency VARCHAR(20) COMMENT '运动频率：从不/偶尔/经常/每天',
    sleep_quality VARCHAR(20) COMMENT '睡眠质量：差/一般/好',
    diet_habits JSON COMMENT '饮食习惯',
    
    -- 画像完整度
    profile_completeness INT DEFAULT 0 COMMENT '画像完整度(0-100)',
    last_profile_update DATETIME COMMENT '最后更新时间',
    profile_version INT DEFAULT 1 COMMENT '画像版本号',
    
    -- AI对话上下文
    preferred_topics JSON COMMENT '偏好话题',
    interaction_style VARCHAR(20) COMMENT '交互风格：简洁/详细',
    
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (user_id) REFERENCES t_user(id),
    INDEX idx_profile_completeness (profile_completeness)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户画像表';

2.3 关联人表 (t_user_relation)

CREATE TABLE t_user_relation (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '关联ID',
    user_id BIGINT NOT NULL COMMENT '用户ID（拥有者）',
    relation_type VARCHAR(20) NOT NULL COMMENT '关系类型：配偶/父母/子女/其他',
    relation_name VARCHAR(50) NOT NULL COMMENT '关系称呼，如"老伴"、"父亲"',
    
    -- 关联人基础信息
    relation_nickname VARCHAR(50) COMMENT '关联人昵称',
    relation_gender TINYINT COMMENT '性别：1-男，2-女',
    relation_birth_date DATE COMMENT '出生日期',
    relation_phone VARCHAR(20) COMMENT '关联人手机号（可选）',
    
    -- 关联人画像ID
    profile_id BIGINT COMMENT '关联人画像ID，关联t_relation_profile',
    
    -- 购买偏好
    is_primary_target TINYINT DEFAULT 0 COMMENT '是否主要购买对象：0-否，1-是',
    purchase_count INT DEFAULT 0 COMMENT '为其购买次数',
    
    status TINYINT DEFAULT 1 COMMENT '状态：0-禁用，1-正常',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (user_id) REFERENCES t_user(id),
    INDEX idx_user_id (user_id),
    INDEX idx_relation_type (relation_type)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='关联人表';

2.4 关联人画像表 (t_relation_profile)

CREATE TABLE t_relation_profile (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '画像ID',
    relation_id BIGINT NOT NULL UNIQUE COMMENT '关联人ID',
    
    -- 基础健康信息（同用户画像）
    height DECIMAL(5,2) COMMENT '身高(cm)',
    weight DECIMAL(5,2) COMMENT '体重(kg)',
    blood_type VARCHAR(5) COMMENT '血型',
    
    -- 健康状况
    chronic_diseases JSON COMMENT '慢性病史',
    allergies JSON COMMENT '过敏史',
    current_medications JSON COMMENT '当前用药',
    health_concerns JSON COMMENT '健康关注点',
    
    -- 画像完整度
    profile_completeness INT DEFAULT 0 COMMENT '画像完整度(0-100)',
    last_profile_update DATETIME COMMENT '最后更新时间',
    
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (relation_id) REFERENCES t_user_relation(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='关联人画像表';

2.5 购买记录表 (t_purchase_record)

CREATE TABLE t_purchase_record (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '购买记录ID',
    order_id VARCHAR(50) NOT NULL COMMENT '订单号',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    
    -- 购买对象
    target_type VARCHAR(20) DEFAULT 'self' COMMENT '购买对象类型：self-本人/relation-关联人',
    target_relation_id BIGINT COMMENT '关联人ID（为他人购买时）',
    
    -- 商品信息
    product_id VARCHAR(50) NOT NULL COMMENT '商品ID',
    product_name VARCHAR(200) NOT NULL COMMENT '商品名称',
    product_category VARCHAR(50) COMMENT '商品分类',
    product_tags JSON COMMENT '商品标签，如["补钙","骨骼健康"]',
    
    -- 购买信息
    quantity INT DEFAULT 1 COMMENT '购买数量',
    unit_price DECIMAL(10,2) COMMENT '单价',
    total_amount DECIMAL(10,2) COMMENT '总金额',
    
    -- 购买场景
    purchase_reason VARCHAR(100) COMMENT '购买原因/健康需求',
    health_symptoms JSON COMMENT '相关症状，如["关节疼","骨质疏松"]',
    
    -- 订单状态
    order_status VARCHAR(20) DEFAULT 'paid' COMMENT '订单状态：paid/shipped/completed/refunded',
    purchase_time DATETIME NOT NULL COMMENT '购买时间',
    
    -- 复购相关
    repurchase_count INT DEFAULT 0 COMMENT '复购次数',
    last_repurchase_time DATETIME COMMENT '最近复购时间',
    next_repurchase_estimate DATE COMMENT '预计复购日期',
    
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (user_id) REFERENCES t_user(id),
    INDEX idx_user_id (user_id),
    INDEX idx_product_id (product_id),
    INDEX idx_purchase_time (purchase_time),
    INDEX idx_target_relation (target_relation_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='购买记录表';

2.6 画像标签表 (t_profile_tag)

CREATE TABLE t_profile_tag (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '标签ID',
    tag_category VARCHAR(50) NOT NULL COMMENT '标签分类：慢性病/过敏源/健康关注点/生活习惯',
    tag_code VARCHAR(50) NOT NULL COMMENT '标签编码',
    tag_name VARCHAR(100) NOT NULL COMMENT '标签名称',
    tag_description VARCHAR(500) COMMENT '标签描述',
    
    -- 标签属性
    is_risk_tag TINYINT DEFAULT 0 COMMENT '是否风险标签：0-否，1-是（如过敏源）',
    risk_level VARCHAR(20) COMMENT '风险等级：低/中/高',
    
    -- 关联商品
    related_product_categories JSON COMMENT '关联商品分类',
    
    -- 追问配置
    follow_up_question VARCHAR(200) COMMENT '追问问题',
    follow_up_options JSON COMMENT '追问选项',
    
    -- 排序和状态
    sort_order INT DEFAULT 0 COMMENT '排序',
    status TINYINT DEFAULT 1 COMMENT '状态：0-禁用，1-正常',
    
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    UNIQUE KEY uk_category_code (tag_category, tag_code),
    INDEX idx_category (tag_category)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='画像标签表';

2.7 画像变更记录表 (t_profile_change_log)

CREATE TABLE t_profile_change_log (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '变更记录ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    profile_id BIGINT NOT NULL COMMENT '画像ID',
    profile_type VARCHAR(20) COMMENT '画像类型：user/relation',
    
    -- 变更信息
    change_type VARCHAR(50) NOT NULL COMMENT '变更类型：新增过敏/新增慢性病/修改基础信息',
    field_name VARCHAR(50) COMMENT '变更字段',
    old_value TEXT COMMENT '旧值',
    new_value TEXT COMMENT '新值',
    
    -- 风险标记
    is_major_change TINYINT DEFAULT 0 COMMENT '是否重大变更：0-否，1-是',
    risk_alert TEXT COMMENT '风险提示',
    
    -- 来源信息
    source VARCHAR(50) COMMENT '来源：对话/用户填写/客服录入',
    conversation_id VARCHAR(50) COMMENT '对话ID（如来自对话）',
    
    -- 处理状态
    review_status VARCHAR(20) DEFAULT 'pending' COMMENT '审核状态：pending/approved/rejected',
    reviewer VARCHAR(50) COMMENT '审核人',
    review_time DATETIME COMMENT '审核时间',
    
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (user_id) REFERENCES t_user(id),
    INDEX idx_user_id (user_id),
    INDEX idx_change_type (change_type),
    INDEX idx_created_at (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='画像变更记录表';

2.8 客服配置表 (t_customer_service)

CREATE TABLE t_customer_service (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '配置ID',
    config_name VARCHAR(50) NOT NULL COMMENT '配置名称',
    
    -- 客服类型
    service_type VARCHAR(20) NOT NULL COMMENT '类型：phone/qrcode/link',
    
    -- 联系方式
    phone_number VARCHAR(20) COMMENT '客服电话',
    qrcode_url VARCHAR(255) COMMENT '二维码图片URL',
    service_link VARCHAR(255) COMMENT '客服链接',
    
    -- 服务时间
    service_hours VARCHAR(100) COMMENT '服务时间，如"工作日 9:00-18:00"',
    
    -- 触发场景
    trigger_scenarios JSON COMMENT '触发场景，如["订单问题","退款问题","投诉"]',
    trigger_keywords JSON COMMENT '触发关键词',
    
    -- 显示配置
    display_text VARCHAR(200) COMMENT '显示文案',
    display_order INT DEFAULT 0 COMMENT '显示顺序',
    
    status TINYINT DEFAULT 1 COMMENT '状态：0-禁用，1-正常',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='客服配置表';

2.9 对话历史表 (t_conversation_history)

CREATE TABLE t_conversation_history (
    id BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '对话ID',
    conversation_id VARCHAR(50) NOT NULL COMMENT '会话ID',
    user_id BIGINT NOT NULL COMMENT '用户ID',
    
    -- 对话信息
    user_message TEXT COMMENT '用户消息',
    ai_response TEXT COMMENT 'AI回复',
    
    -- 意图信息
    intent_type VARCHAR(50) COMMENT '意图类型',
    intent_confidence DECIMAL(5,4) COMMENT '意图置信度',
    
    -- 画像追问
    profile_questions_asked JSON COMMENT '追问的画像问题',
    profile_answers_received JSON COMMENT '用户回答的画像信息',
    
    -- 产品推荐
    products_recommended JSON COMMENT '推荐的产品列表',
    product_clicked VARCHAR(50) COMMENT '用户点击的产品',
    
    -- 响应时间
    response_time_ms INT COMMENT '响应时间(毫秒)',
    
    -- 意图确认
    intent_confirm_triggered TINYINT DEFAULT 0 COMMENT '是否触发意图确认',
    intent_confirm_result VARCHAR(20) COMMENT '意图确认结果：confirmed/switched/cancelled',
    
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (user_id) REFERENCES t_user(id),
    INDEX idx_conversation_id (conversation_id),
    INDEX idx_user_id (user_id),
    INDEX idx_created_at (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='对话历史表';

4.2 查询优化建议

-- 查询用户购买记录（按时间倒序）
SELECT * FROM t_purchase_record 
WHERE user_id = ? AND order_status = 'completed'
ORDER BY purchase_time DESC
LIMIT 10;

-- 查询用户画像完整度
SELECT user_id, profile_completeness 
FROM t_user_profile 
WHERE profile_completeness < 50;

-- 查询需要复购提醒的用户
SELECT user_id, product_id, product_name, next_repurchase_estimate
FROM t_purchase_record
WHERE next_repurchase_estimate <= CURDATE() + INTERVAL 7 DAY
AND order_status = 'completed';
